-RUTA:
C:\xampp\apache\conf\extra\httpd-vhosts.conf

-ABRIR CON BLOC DE NOTAS

-PEGAR ESTO AL ULTIMO DEL ARCHIVO

NameVirtualHost *:80
<VirtualHost *:80>
    ServerName empaquesys.local
    DocumentRoot "C:/xampp/htdocs/empaquesys/public"
    <Directory "C:/xampp/htdocs/empaquesys/public">
 	Options Indexes FollowSymLinks MultiViews
            	AllowOverride All
            	Order allow,deny
            	allow from all
   </Directory>
</VirtualHost>
